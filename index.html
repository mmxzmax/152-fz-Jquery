"personalAgreementUrl":"", url страницы с соглашением
    "personalAgreementLink":"", текст ссылки на стрнаицу
    "personalAgreementText":"", текст чекбокса перед ссылкой
    "beforeSelector":'',        селектор элемента перед которым будет вставлен чекбокс
    "hideSubmitButton":false    срывать ли кнопку отправки формы если пользователель не отметил флажок


    пример использования

    $(document).ready(function(){
        var settings={
            personalAgreementUrl:"/",
            personalAgreementText:"подтверждаю согласие на обработку ",
            personalAgreementLink:"персональных данных",
            hideSubmitButton:true
        };
       $('#form1').personalDataCheckBox(settings);
    });
